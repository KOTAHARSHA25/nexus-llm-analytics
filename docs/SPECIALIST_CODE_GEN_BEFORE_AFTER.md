### StatisticalAgent - After

```python
class StatisticalAgent(BasePluginAgent):
    def initialize(self, **kwargs) -> bool:
        # Existing config
        self.confidence_level = self.config.get("confidence_level", 0.95)
        self.alpha = 1 - self.confidence_level
        
        # NEW: Code generation support
        self._orchestrator = None
        self.initializer = get_model_manager()
    
    def execute(self, query: str, data: Any = None, **kwargs):
        # 1. Intelligent routing via orchestrator
        plan = self._orchestrator.create_execution_plan(
            query=query,
            context={'agent': self.get_metadata().name},
            llm_client=self.initializer.llm_client
        )
        
        # 2. Route to code generation OR standard methods
        if plan.execution_method == ExecutionMethod.CODE_GENERATION:
            # NEW: Generate custom Python code
            result, metadata = self._execute_with_code_gen(...)
            return {"success": True, "result": result, "metadata": metadata}
        
        # 3. Fallback to proven deterministic methods
        intent = self._parse_statistical_intent(query)
        if intent == "ttest":
            return self._t_test_analysis(...)
        # ... all existing methods still work
    
    def _execute_with_code_gen(self, query, data, model, filepath):
        """NEW METHOD: Generate and execute Python code for complex queries"""
        code_gen = get_code_generator()
        result = code_gen.generate_and_execute(query, df=data, model=model)
        return formatted_result, metadata
```

**Capabilities:**
- ✅ Handles ALL queries (simple + complex)
- ✅ Can perform novel statistical analyses via code generation
- ✅ Fixed methods PLUS unlimited flexibility
- ✅ Complex queries get custom solutions
- ✅ Backward compatible with all existing functionality

## Code Generation Process

### Step-by-Step: _execute_with_code_gen()

```python
def _execute_with_code_gen(self, query, data, model, filepath):
    # 1. Ensure DataFrame available
    if not isinstance(data, pd.DataFrame):
        data = pd.read_csv(filepath)
    
    # 2. Get code generator
    code_gen = get_code_generator()
    
    # 3. Generate Python code using LLM
    result = code_gen.generate_and_execute(
        query=query,
        df=data,
        model=model,
        analysis_context={'agent': 'StatisticalAgent'}
    )
    
    # 4. Check success
    if result.success:
        # Format result nicely
        display_text = format_result(result.result)
        metadata = {
            "execution_method": "code_generation",
            "generated_code": result.generated_code,
            "execution_time_ms": result.execution_time_ms
        }
        return display_text, metadata
    else:
        # Failure triggers fallback to standard methods
        raise Exception(f"Code gen failed: {result.error}")
```
